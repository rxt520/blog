class QueueArr{
    private int maxSize;
    private int front;
    private int rear;
    private int[] arr;
    public QueueArr(int maxSize) {
        this.maxSize = maxSize;
        front = -1;
        rear = -1;
        arr = new int[maxSize];
    }

    public boolean isFull(){
        if ((rear+1)%maxSize == front) return true;
        return false;
    }

    public boolean isEmpty(){
        if (front==-1 && rear == -1){
            return true;
        }
        return false;
    }

    public void addQueue(int data){
        if (isFull()){
            System.out.println("队列空间已满");
        }else {
            if (front==-1){
                front = 0;
            }
            if (rear==maxSize-1){
                rear = 0;
            }else {
                int crear = rear ;
                crear++;
                if (rear!=-1 && crear%maxSize==front){
                    System.out.println("队列空间已满");
                    return;
                }else {
                    rear++;
                }
            }
            arr[rear] = data;
        }
    }

    public int getQueue(){
        if (isEmpty()){
            return -1;
        }
        else if (front==rear){
            int pop =  arr[front];
            front=-1;
            rear=-1;
            arr = new int[maxSize];
            return pop;
        }
        else {
            int pop =  arr[front];
            front = ++front%maxSize;
            return pop;
        }
    }
}

        arr[left] = arr[i];
        arr[i] = temp;

        quickSort(arr,i+1,right);

# 插值查找

  • mid = left + (condition-arr[left])/(arr[right]-arr[left])*(right-left)
  • 原理 查找 该条件数字在索引下标的相对位置，如果不是则递归
package com.rxt.algorithm.search;

/**
 * @author Ruan.X.T.
 * @create 2020-03-08-14:45
 */
public class InsertValueSearch {
    public static void main(String[] args) {
        int[] arr = {1,15,30,40,45,50,55,60,65,70,75};
        int i = insertValueSearch(arr, 0, arr.length - 1, 55);
        System.out.println(i);
    }
    public static int insertValueSearch(int[] arr,int left,int right,int condition){
        if (left>right || condition<arr[left] || condition>arr[right] ){
            return -1;
        }
        int mid = left + (condition-arr[left])/(arr[right]-arr[left])*(right-left);
        if (arr[mid]==condition){
            return mid;
        }else if (arr[mid]<condition){
            return insertValueSearch(arr,mid+1,right,condition);
        }else {
            return insertValueSearch(arr,left,mid-1,condition);
        }
    }
}

package com.rxt.algorithm.search;

import java.util.Arrays;

/**
 * @author Ruan.X.T.
 * @create 2020-03-08-18:34
 */
public class FibonacciSearch {
    static int fiboSize = 20;
    public static void main(String[] args) {
        int[] arr = {5,7};
        int i = fibonaciiSearch(arr, 7);
        System.out.println(i);
    }
    public static int fibonaciiSearch(int[] arr,int condition){
        int low = 0;
        int high = arr.length-1;
        int[] fib = fibonacii();
        int k = 0;
        while (fib[k]-1<high){
            k++;
        }
        int[] temp = Arrays.copyOf(arr,fib[k]);
        for (int i = high+1; i < temp.length; i++) {
            temp[i] = arr[high];
        }
        while (low<=high){
            int mid =low+fib[k-1]-1;
            if (temp[mid]==condition){
                if (mid<high){
                    return mid;
                }else {
                    return high;
                }
            }else if (temp[mid]<condition){
                low = mid + 1;
                k--;
            }else {
                high = mid - 1;
                k -= 2;
            }
        }
        return -1;
    }
    public static int[] fibonacii(){
        int[] arr = new int[fiboSize];
        arr[0] = 1;
        arr[1] = 1;
        for (int i = 2; i < fiboSize ; i++) {
            arr[i] = arr[i-1] + arr[i-2];
        }
        return arr;
    }
}
